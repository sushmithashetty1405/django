#views
from datetime import datetime
from django.shortcuts import render


# Using datetime library
def current_datetime(request):
    print("current", request.build_absolute_uri())
    now = datetime.now()
    return render(request, "datetimeapp/current_datetime.html", {'current_date_time': now})
